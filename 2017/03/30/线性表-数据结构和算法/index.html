线性表 - 数据结构和算法

2017-03-30

线性表两种物理存储结构：顺序存储结构和链式存储结构

顺序存储结构

静态顺序表

(1)存和读 O(1)  
(2)插入和删除 O(n)  
(3)适用于不经常插入和删除元素的存储

结构定义

#define MAXSIZE 20
typedef int ElemType;
typedef struct {
ElemType data[MAXSIZE];    // 存储空间
int length;    // 当前元素个数
} SqList;

用e返回L中第i个元素的值

Status GetElem(SqList L, int i, ElemType *e) 
{    
    if (L.length == 0 || i < 1 || i > L.length) {
        return ERROR;
    }
    *e = L.data[i-1];
    return OK;
}

在顺序表第i个位置之前插入元素e

Status ListInsert(SqList *L, int i, ElemType e)
{    
    if (L.length == MAXSIZE) {
        return ERROR; // 已满，插不了
    }

    if (i < 1 || i > L.length) {
        return ERROR; // i不符合条件 
    }

    if (i <= L.length) {    //插入位置不在表尾
        for (int j = L.length; j > i-1; j--) {
                L.data[j] = L.data[j - 1];
        }
    }

    L.data[i - 1] = e;
    L.length++;

    return OK;
}

删除顺序表L的第i个元素结点，并用e返回其值

Status ListDelete(SqList *L, int i, ElemType *e)
{    
    if (L.length == 0) {
        return ERROR; //没有元素
    }

    if (i < 1 || i > L.length) {
        return ERROR; // i不符合条件 
    }

     *e = L.data[i-1];

    if (i <= L.length) {    //插入位置不在表尾
        for (int j = i; j < L.length; j++) {
                L.data[j - 1] = L.data[j];
        }
    }

    L.length--;

    return OK;
}


链式存储结构

带头结点的单链表

1.适用于频繁插入、删除结点
2.事先不知道线性表的长度

结构定义

typedef int ElemType;
typedef struct Node {
    ElemType data; //数据域
    struct Node* next; //指针域
} Node;
typedef struct Node* LinkList;

获取链表L的第i个元素的值，赋给e

Status GetElem(LinkList L, int i, ElemType *e) 
{    
    int j = 1;
    LinkList p = L->next;

    while (p && j<i) {
        p = p->next;
        j++;
    }

    if (!p || j>i) {
        return ERROR;
    }

    *e = p->data;
    return OK;
}

在链表L的第i个位置之前插入新的元素e

Status ListInsert(LinkList *L, int i, ElemType e) 
{    
    int j = 1;
    LinkList p, s;
    p = （*L)->next;

    while (p && j<i-1) {  // 找到第i-1个结点
        p = p->next;
        j++;
    }

    if (!p || i<1) {
        return ERROR;
    }

    s = (LinkList)malloc(sizeof(Node));
    s->data = e;

    s->next = p->next;    // 将s插到第i-1个结点后，即第i个结点之前
    p->next = s;

    return OK;
}

删除链表L的第i个元素，L的长度减1

Status ListDelete(LinkList *L, int i, ElemType *e) 
{    
    int j = 1;
    LinkList p, q;
    p = (*L)->next;

    while (p && j<i-1) { // 找到第i-1个结点
        p = p->next;
        j++;
    }

    if (!p || !(p->next) || i<1 ) {
        return ERROR;
    }

    q = p->next;
    p->next = q->next;

    *e = q->data;
    free(q);

    return OK;
}

头插法建立单链表

void CreateListHead(LinkList *L, int n) 
{        LinkList p;

    *L = (LinkList)malloc(sizeof(Node));
    (*L)->next = NULL;

    for (int i = 0; i < n; ++i) {
        p = (LinkList)malloc(sizeof(Node));
        p->data = rand() % 100 + 1;   // 生成0-100的随机数

        p->next = (*L)->next;
        (*L)->next = p;
    }

}

尾插法建立单链表

void CreateListTail(LinkList *L, int n) 
{    
    LinkList p, r;

    *L = (LinkList)malloc(sizeof(Node));
    r = *L;  // 空表，所以尾指针指向头指针

    for (int i = 0; i < n; ++i) {
        p = (LinkList)malloc(sizeof(Node));
        p->data = rand() % 100 + 1;   // 生成0-100的随机数

        r->next = p;
        r = p;
    }

}

清空单链表

Status ClearList(LinkList *L)
{    
    LinkList p, q;
    p = (*L)->next;

    while (p) {
        q = p->next;
        free(p);
        p = q;
    }

    (*L)->next = NULL;

    return OK;
}


静态链表

1.数组的第一个和最后一个元素的data不存放数据；
2.把未使用的数组元素称为备用链表
3.数组的第一个元素，即下标为0的那个元素的cur存放备用链表的第一个结点的下标
4.数组的最后一个元素，即下标为MAXSIZE-1的cur则存放第一个有数值的元素的下标，相当于但链表的头结点

结构定义

#define MAXSIZE 1000
typedef struct {
    ElemType data; //数据
    int cur;        //游标cursor
}Component, StaticLinkList[MAXSIZE];

初始化静态链表，相当于初始化数组

Status InitList(StaticLinkList space)
{    
    for (int i = 0; i < MAXSIZE-1; ++i) {
        space[i].cur = i+1;
    }
    space[MAXSIZE-1].cur = 0;

    return OK;
}

获取空闲分量的下标

int Malloc_SLL(StaticLinkList space)
{
    int i = space[0].cur; 
    if (space[0].cur) {
        space[0].cur = space[i].cur;
        //把它的下一个分量用来作为备用
    }
    return i;
}

在静态链表L中第i个元素之前插入新的元素e

Status ListInsert(StaticLinkList L, int i, ElemType e)
{
    int k = MAXSIZE - 1;  // 数组的最后一个元素

    if (int i < 1 || i > ListLength(L) + 1) {
        return ERROR;
    }

    int j = Malloc_SLL(L);    // 获取第一个空闲的分量的下标

    if (j) {
        L[j].data = e;
        for (int l = 1; l <= j - 1; l++) {
            k = L[k].cur;
        }
        L[j].cur = L[k].cur;
        L[k].cur = j;

        return OK;
    }

    return OK;
}


循环链表

1.将单链表中终端结点的指针端由空指针改为指向头结点，就使整个单链表形成一个环，这种头尾相接的单链表成为单循环链表，简称循环链表。
2.循环链表的单链表的主要差异就在于循环的判断空链表的条件上，原来判断head->next是否为null，现在则是head->next是否等于head。
